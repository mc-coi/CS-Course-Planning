## Example: Adding File Save Feature

```python
def save_report(filename, name, scores, average, grade):
    """
    Save grade report to a text file.

    Parameters:
        filename (str): Output file path
        name, scores, average, grade: Report data

    Returns:
        bool: True if successful, False if error
    """
    try:
        with open(filename, "w") as f:
            f.write("=" * 40 + "\n")
            f.write(f"Grade Report: {name}\n")
            f.write("=" * 40 + "\n")
            f.write(f"Scores: {scores}\n")
            f.write(f"Average: {average:.1f}\n")
            f.write(f"Grade: {grade}\n")
            f.write("=" * 40 + "\n")
        print(f"✓ Report saved to {filename}")
        return True
    except IOError as e:
        print(f"✗ Error saving file: {e}")
        return False

# Test it:
save_report("alice_report.txt", "Alice", [85, 92, 78], 85.0, "B")
```
