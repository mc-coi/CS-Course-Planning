### Example: Grade Processing System

```python
def filter_passing_grades(grades):
    """
    Return only grades that are 70 or higher.

    Parameters:
    grades (list): List of numerical grades

    Returns:
    list: Grades >= 70
    """
    passing = []
    for grade in grades:
        if grade >= 70:
            passing.append(grade)
    return passing

def calculate_class_average(grades):
    """
    Calculate the average of all grades.

    Parameters:
    grades (list): List of numerical grades

    Returns:
    float: Average grade, or 0 if list empty
    """
    if not grades:
        return 0
    return sum(grades) / len(grades)

def count_students_above(grades, threshold):
    """
    Count how many students scored above a threshold.

    Parameters:
    grades (list): List of numerical grades
    threshold (float): Score threshold

    Returns:
    int: Number of students above threshold
    """
    count = 0
    for grade in grades:
        if grade > threshold:
            count = count + 1
    return count

def get_grade_distribution(grades):
    """
    Count grades in each letter category.

    Parameters:
    grades (list): List of numerical grades

    Returns:
    dict: Counts of A, B, C, D, F
    """
    dist = {'A': 0, 'B': 0, 'C': 0, 'D': 0, 'F': 0}
    for grade in grades:
        if grade >= 90:
            dist['A'] += 1
        elif grade >= 80:
            dist['B'] += 1
        elif grade >= 70:
            dist['C'] += 1
        elif grade >= 60:
            dist['D'] += 1
        else:
            dist['F'] += 1
    return dist

def identify_top_performers(grades, top_n):
    """
    Return the top N grades.

    Parameters:
    grades (list): List of numerical grades
    top_n (int): How many top grades to return

    Returns:
    list: The N highest grades, sorted descending
    """
    sorted_grades = sorted(grades, reverse=True)
    return sorted_grades[:top_n]

# Main program
if __name__ == "__main__":
    grades = [85, 92, 78, 91, 68, 88, 95, 72, 81, 79, 90, 87]

    print("=== Grade Analysis ===")
    print(f"All grades: {grades}")

    passing = filter_passing_grades(grades)
    print(f"Passing grades (70+): {passing}")

    avg = calculate_class_average(grades)
    print(f"Class average: {avg:.2f}")

    above_90 = count_students_above(grades, 90)
    print(f"Students with 90+: {above_90}")

    dist = get_grade_distribution(grades)
    print(f"Grade distribution: {dist}")

    top_3 = identify_top_performers(grades, 3)
    print(f"Top 3 grades: {top_3}")
```
