### Example: Student Grade Manager

```python
def add_grade(grades_list, grade):
    """
    Adds a grade to the list.

    Parameters:
    grades_list (list): List of grades
    grade (float): New grade to add (0-100)

    Returns:
    list: Updated list of grades
    """
    grades_list.append(grade)
    return grades_list

def calculate_average(grades):
    """
    Calculates the average of a list of grades.

    Parameters:
    grades (list): List of numerical grades

    Returns:
    float: The average grade, or 0 if list is empty
    """
    if not grades:
        return 0
    return sum(grades) / len(grades)

def get_letter_grade(average):
    """
    Converts a numerical average to a letter grade.

    Parameters:
    average (float): Numerical grade (0-100)

    Returns:
    str: Letter grade (A, B, C, D, or F)

    Grading scale:
    A: 90-100, B: 80-89, C: 70-79, D: 60-69, F: 0-59
    """
    if average >= 90:
        return 'A'
    elif average >= 80:
        return 'B'
    elif average >= 70:
        return 'C'
    elif average >= 60:
        return 'D'
    else:
        return 'F'

def print_report(name, grades):
    """
    Displays a complete grade report for a student.

    Parameters:
    name (str): Student's name
    grades (list): List of numerical grades

    Returns:
    None (prints to console)
    """
    average = calculate_average(grades)
    letter = get_letter_grade(average)
    print(f"\n=== Grade Report for {name} ===")
    print(f"Grades: {grades}")
    print(f"Average: {average:.2f}")
    print(f"Letter Grade: {letter}")

# Main program
if __name__ == "__main__":
    grades = []

    # Add some grades
    grades = add_grade(grades, 85)
    grades = add_grade(grades, 92)
    grades = add_grade(grades, 78)
    grades = add_grade(grades, 88)

    # Display report
    print_report("Alice", grades)
    print_report("Bob", [95, 87, 91, 89])
```
